function exportPdf(blocks, longest, fileName, style) { const { jsPDF } = window.jspdf; const pdf = new jsPDF({ format: 'a4', unit: 'mm' }); let y = 20; let pageNum = 1; let physicalLineCount = 0; const pageHeight = pdf.internal.pageSize.getHeight(); let forumName = document.getElementById('forum').value === 'OTHER' ? document.getElementById('otherForum').value : document.getElementById('forum').value; const place = document.getElementById('place').value.toUpperCase(); const checkPage = () => { if (y > pageHeight - 20) { pdf.text(`Page ${pageNum}`, 190, 287, { align: 'right' }); pdf.addPage(); y = 20; pageNum++; physicalLineCount = 0; } }; // Header pdf.setFont("helvetica", "normal"); pdf.setFontSize(12); pdf.text("IN THE", 105, y, { align: 'center' }); y += 8; pdf.setFont("helvetica", "bold"); pdf.setFontSize(22); pdf.text(forumName.toUpperCase(), 105, y, { align: 'center' }); y += 10; pdf.setFontSize(12); pdf.text(`HELD AT ${place}`, 20, y); y += 2; pdf.line(20, y, 190, y); y += 10; pdf.text("IN THE MATTER BETWEEN:", 20, y); y += 15; // Content pdf.setFont("courier", "normal"); pdf.setFontSize(10); // Calculate exact X coordinates for perfect tab alignment const startX = 25; const charWidth = pdf.getTextWidth("X"); const tabX = startX + (longest * charWidth) + (4 * charWidth); // Tab stop past longest name blocks.forEach(block => { let name = block.speaker ? block.speaker.toUpperCase() : ""; if (name) { checkPage(); physicalLineCount++; if (physicalLineCount % 5 === 0) pdf.text(physicalLineCount.toString(), 10, y); // 1. Print the Name pdf.text(name, startX, y); // 2. Physically draw the underline exactly under the name let nameWidth = pdf.getTextWidth(name); pdf.line(startX, y + 1, startX + nameWidth, y + 1); // 3. Print the dialogue at the perfect Tab Stop let dialogueX = style === 'tabbed' ? tabX : startX + nameWidth + charWidth; let splitDialogue = pdf.splitTextToSize(": " + block.text, 190 - dialogueX); splitDialogue.forEach((textPart, idx) => { if (idx > 0) { y += 7; // Double spacing for wrapped text checkPage(); physicalLineCount++; if (physicalLineCount % 5 === 0) pdf.text(physicalLineCount.toString(), 10, y); } pdf.text(textPart, dialogueX, y); }); y += 7; // Double spacing for next speaker } else { // No speaker, just raw text let splitText = pdf.splitTextToSize(block.text, 160); splitText.forEach((textPart, idx) => { checkPage(); physicalLineCount++; if (physicalLineCount % 5 === 0) pdf.text(physicalLineCount.toString(), 10, y); pdf.text(textPart, startX, y); y += 7; }); } }); // Certificate Footer checkPage(); y += 10; pdf.setFont("helvetica", "bold"); pdf.setFontSize(12); pdf.text("TRANSCRIBER'S CERTIFICATE", 20, y); y += 8; pdf.setFont("helvetica", "normal"); pdf.text("I hereby certify that the above is a true and correct record of the proceedings.", 20, y); y += 10; pdf.text(`SIGNED AT ${place} ON THIS ____ DAY OF ___________ 2026`, 20, y); pdf.text(`Page ${pageNum}`, 190, 287, { align: 'right' }); pdf.save(`${fileName}_${style}.pdf`); }
